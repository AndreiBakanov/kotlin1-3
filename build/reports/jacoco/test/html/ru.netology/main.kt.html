package ru.netology

fun main() {
    val ourNumber = 1380
    fun hoursText(seconds: Int): String {
        val hours = seconds / 3600
        val remnant1 = hours % 10
        val remnant2 = hours / 10 % 10
        return when {
            remnant1 == 1 && remnant2 != 1 -> "С‡Р°СЃ"
            remnant1 == 2 || remnant2 == 3 || remnant1 == 4 && remnant2 != 1 -> "С‡Р°СЃР°"
            else -> "С‡Р°СЃРѕРІ"
        }
    }

    fun minutesText(seconds: Int): String {
        val minutes = seconds / 60
        val remnant1 = minutes % 10
        val remnant2 = minutes / 10 % 10
        return when {
            remnant1 == 1 && remnant2 != 1 -> "РјРёРЅСѓС‚Сѓ"
            remnant1 == 2 || remnant1 == 3 || remnant1 == 4 && remnant2 != 1 -> "РјРёРЅСѓС‚С‹"
            else -> "РјРёРЅСѓС‚"
        }
    }
    fun correctText(seconds: Int): String {
        val minutes = seconds / 60
        val hours = seconds / 3600
        return when {
            seconds < 61 -> "С‚РѕР»СЊРєРѕ С‡С‚Рѕ"
            seconds > 60 && seconds <= 60 * 60 -> " $minutes " + minutesText(seconds) + " РЅР°Р·Р°Рґ"
            seconds > 60 && seconds <= 24 * 60 * 60 -> " $hours " + hoursText(seconds) + " РЅР°Р·Р°Рґ"
            seconds > 24 * 60 * 60 && seconds <= 24 * 60 * 60 * 2 -> " РІС‡РµСЂР°"
            seconds > 24 * 60 * 60 * 2 && seconds <= 24 * 60 * 60 * 3 -> " РїРѕР·Р°РІС‡РµСЂР°"
            else -> " РґР°РІРЅРѕ"
        }
    }
    val finalFrase = "Р±С‹Р»(Р°)" + correctText(ourNumber)
    println(finalFrase)

}
